<script>
function openModal_mentors() {
  document.getElementById("mentors_modal").style.display = "block";
}

function closeModal_mentors() {
  document.getElementById("mentors_modal").style.display = "none";
}

var slideIndex = 1;
showSlides_mentor(slideIndex);

function plusSlides_mentors(n) {
  showSlides_mentor(slideIndex += n);
}

function currentSlide_mentors(n) {
  showSlides_mentor(slideIndex = n);
}

function showSlides_mentor(n) {
  var i;
  var slides_mentors = document.getElementsByClassName("mySlides_mentors");
  var dots_mentors = document.getElementsByClassName("demo_mentors");
  var captionText_mentors = document.getElementById("caption_mentors");
  if (n > slides_mentors.length) {slideIndex = 1}
  if (n < 1) {slideIndex = slides_mentors.length}
  for (i = 0; i < slides_mentors.length; i++) {
      slides_mentors[i].style.display = "none";
  }
  for (i = 0; i < dots_mentors.length; i++) {
      dots_mentors[i].className = dots_mentors[i].className.replace(" active", "");
  }
  slides_mentors[slideIndex-1].style.display = "block";
  dots_mentors[slideIndex-1].className += " active";
  captionText_mentors.innerHTML = dots_mentors[slideIndex-1].alt;
}
</script>

<script>
var modal_mentors = document.getElementById('mentors_modal');

window.onclick = function(event_mentors) {
  if (event_mentors.target == modal_mentors) {
    modal_mentors.style.display = "none";
  }
}
</script>
